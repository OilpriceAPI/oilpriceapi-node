Quick Start

import { OilPriceAPI } from '@oilpriceapi/client';

// Initialize client
const client = new OilPriceAPI({
  apiKey: process.env.OILPRICEAPI_KEY
});

// Get latest Brent Crude price
const brent = await client.getLatestPrice('BRENT_CRUDE_USD');
console.log(`Brent Crude: $${brent.price.toFixed(2)}`);
// Output: Brent Crude: $71.45

// Get historical data
const history = await client.getPastYear('WTI_USD');
console.log(`${history.data.length} days of WTI data`);

// Get multiple prices at once
const prices = await client.getMultiplePrices([
  'BRENT_CRUDE_USD',
  'WTI_USD',
  'NATURAL_GAS_USD'
]);
prices.forEach(p => {
  console.log(`${p.code}: $${p.price.toFixed(2)}`);
});
